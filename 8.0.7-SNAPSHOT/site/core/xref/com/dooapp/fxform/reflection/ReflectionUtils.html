31      /**
32       * Tries to retrieve the generic parameter of an ObjectProperty at runtime.
33       */
34      public static Class getObjectPropertyGeneric(Object source, Field field) {
35          Type type = field.getGenericType();
36          if (type instanceof ParameterizedType) {
37              return getGenericClass(source, (ParameterizedType) type);
38          } else {
39              return field.getType();
40          }
41      }

43      /**
44       * Tries to retrieve the generic parameter of an ObjectProperty return by a method at runtime.
45       */
46      public static Class getMethodReturnTypeGeneric(Object source, Method method) {
47          Type type = method.getGenericReturnType();
48          if (type instanceof ParameterizedType) {
49              return getGenericClass(source, (ParameterizedType) type);
50          } else {
51              return method.getReturnType();
52          }
53      }

55      /**
56       * Try to extract the generic type of the given ParameterizedType used in the given source object.
57       *
58       * @param source
59       * @param type
60       * @return
61       */
62      private static Class getGenericClass(Object source, ParameterizedType type) {
63          Type type1 = type.getActualTypeArguments()[0];
64          if (type1 instanceof ParameterizedType) {
65              return (Class) ((ParameterizedType) type1).getRawType();
66          } else if (type1 instanceof TypeVariable) {
67              // Type is generic, try to get its actual type from the super class
68              // e.g.: ObjectProperty<T> where T extends U
69              if (source.getClass().getGenericSuperclass() instanceof ParameterizedType) {
70                  Type parameterizedType = ((ParameterizedType) source.getClass().getGenericSuperclass()).getActualTypeArguments()[0];
71                  if (parameterizedType instanceof ParameterizedType) {  // it means that the parent class is also generic
72                      return (Class) ((ParameterizedType) parameterizedType).getRawType();
73                  } else {
74                      return (Class) parameterizedType;
75                  }
76              } else {
77                  // The actual type is not declared, use the upper bound of the type e.g. U
78                  return (Class) ((TypeVariable) type1).getBounds()[0];
79              }
80          } else {
81              return (Class) type1;
82          }
83      }

97      /**
98       * Extract a collection of fields using their name from a given class and fill
99       * a list with it.
100      *
101      * @param clazz  the Class to extract the fields from
102      * @param result the list of Fields to fill with the extracted fields
103      * @param fields the field names
104      */
105     public static void fillFieldsByName(Class clazz, List<Field> result, String... fields) {
106         for (String fieldName : fields) {
107             Field field = getFieldByName(clazz, fieldName);
108             if (field != null) {
109                 result.add(field);
110             }
111         }
112     }

114     /**
115      * Extract a Field from a given class using its name.
116      * This method will look into the given class declared fields. If the field is not
117      * found, it will then look recursively in all super classes of the given class.
118      *
119      * @param clazz     the class to extract the Field from
120      * @param fieldName the name of the field to extract
121      * @return the extracted Field, null if not found
122      */
123     public static Field getFieldByName(Class clazz, String fieldName) {
124         try {
125             Field field = clazz.getDeclaredField(fieldName);
126             fixAccessible(field);
127             return field;
128         } catch (NoSuchFieldException e) {
129             if (clazz.getSuperclass() != null && clazz.getSuperclass() != Object.class) {
130                 return getFieldByName(clazz.getSuperclass(), fieldName);
131             }
132         }
133         return null;
134     }

136     /**
137      * Get a list of all fields from a given class. This includes all declared fields from the given class and its super classes.
138      *
139      * @param clazz the class to get the fields from
140      * @return the list of all declared fields
141      */
142     public static List<Field> listFields(Class clazz) {
143         List<Field> result = new LinkedList<Field>();
144         ReflectionUtils.fillFields(clazz, result);
145         return result;
146     }
147 
148     /**
149      * Fill a list with all fields from a given class.This includes all declared fields from the given class and its super classes.
150      *
151      * @param clazz  the class to get the fields from
152      * @param result the list to fill
153      */
154     public static void fillFields(Class clazz, List<Field> result) {
155         for (Field field : clazz.getDeclaredFields()) {
156             result.add(field);
157         }
158 
159         fixAccessible(result);
160         if (clazz.getSuperclass() != null && clazz.getSuperclass() != Object.class) {
161             fillFields(clazz.getSuperclass(), result);
162         }
163     }

165     /**
166      * Fix the accessibility of a collection of fields to true
167      *
168      * @param result the collection of Fields to fix
169      */
170     private static void fixAccessible(List<Field> result) {
171         for (Field field : result) {
172             fixAccessible(field);
173         }
174     }
175 
176     /**
177      * Fix the accessibility of a field to true
178      *
179      * @param field the fields to fix
180      */
181     private static void fixAccessible(Field field) {
182         if (!field.isAccessible()) {
183             field.setAccessible(true);
184         }
185     }
