16  import com.dooapp.fxform.reflection.MultipleBeanSource;

18  import java.lang.reflect.Field;
19  import java.util.ArrayList;
20  import java.util.LinkedList;
21  import java.util.List;

23  /**
24   * Abstract base class for field providers that add support for {@link com.dooapp.fxform.reflection.MultipleBeanSource}
25   * <p/>
26   * User: Antoine Mischler <antoine@dooapp.com>
27   * Date: 16/12/2013
28   * Time: 15:00
29   */
30  public abstract class MultipleSourceFieldProvider implements FieldProvider {
31  
32      @Override
33      public List<Field> getProperties(Object source) {
34          List<Field> result = new LinkedList<Field>();
35          if (source != null) {
36              if (source instanceof MultipleBeanSource) {
37                  MultipleBeanSource multipleBeanSource = (MultipleBeanSource) source;
38                  for (Object s : multipleBeanSource.getSources()) {
39                      getFields(result, s.getClass());
40                  }
41              } else {
42                  getFields(result, source.getClass());
43              }
44          }
45          return result;
46      }
47  
48      @Override
49      public List<Field> getProperties(Object source, List<String> fields) {
50          List<Field> result = new LinkedList<Field>();
51          if (source != null) {
52              if (source instanceof MultipleBeanSource) {
53                  MultipleBeanSource multipleBeanSource = (MultipleBeanSource) source;
54                  getMultipleFields(result, multipleBeanSource, fields);
55              } else {
56                  getFields(result, source.getClass(), fields);
57              }
58          }
59          return result;
60      }
61  
62      /**
63       * This method tries to extract the fields from the right source in the multiple source bean.
64       *
65       * @param result
66       * @param source
67       * @param fields
68       */
69      protected void getMultipleFields(List<Field> result, MultipleBeanSource source, List<String> fields) {
70          List<String> notSpecificFields = new ArrayList<String>();
71          List<SpecificField> specificFields = new ArrayList<SpecificField>();
72          for (String field : fields) {
73              String[] fieldValues = field.split("-");
74              if (fieldValues.length == 2) {
75                  specificFields.add(new SpecificField(fieldValues[0], fieldValues[1]));
76              } else {
77                  notSpecificFields.add(field);
78              }
79          }
80          for (SpecificField specificField : specificFields) {
81              for (Object s : source.getSources()) {
82                  if (s.getClass().getName().equals(specificField.getClassName())) {
83                      List<String> includeList = new ArrayList<String>();
84                      includeList.add(specificField.getName());
85                      getFields(result, s.getClass(), includeList);
86                  }
87              }
88          }
89          if (!notSpecificFields.isEmpty()) {
90              for (Object s : source.getSources()) {
91                  getFields(result, s.getClass(), notSpecificFields);
92              }
93          }
94      }
95  
96      private class SpecificField {
97  
98          private String className;
99  
100         private String name;
101 
102         private SpecificField(String className, String name) {
103             this.className = className;
104             this.name = name;
105         }
106 
107         public String getClassName() {
108             return className;
109         }
110 
111         public String getName() {
112             return name;
113         }
114     }
115 
116     protected abstract void getFields(List<Field> result, Class aClass);
117 
118     protected abstract void getFields(List<Field> result, Class aClass, List<String> includes);
