33  public class PropertyElementController<WrappedType> extends ElementController<WrappedType> {
34  
35      private final NodeController constraintController;
36  
37      public PropertyElementController(final AbstractFXForm fxForm, PropertyElement element) {
38          super(fxForm, element);
39          PropertyElementValidator validator = ((PropertyEditorController) editorController).getPropertyElementValidator();
40          validator.constraintViolationsProperty().addListener(new ListChangeListener<ConstraintViolation>() {
41              @Override
42              public void onChanged(Change<? extends ConstraintViolation> change) {
43                  while (change.next()) {
44                      fxForm.getConstraintViolations().addAll(change.getAddedSubList());
45                      fxForm.getConstraintViolations().removeAll(change.getRemoved());
46                  }
47              }
48          });
49          constraintController = new ConstraintController(fxForm, element, validator.constraintViolationsProperty());
50          updateSkin((FXFormSkin) fxForm.getSkin());
51          validator.invalidProperty().addListener(new ChangeListener<Boolean>() {
52              @Override
53              public void changed(ObservableValue<? extends Boolean> observableValue, Boolean aBoolean, Boolean aBoolean2) {
54                  if (aBoolean2) {
55                      addStyle(FXForm.INVALID_STYLE);
56                  } else {
57                      removeStyle(FXForm.INVALID_STYLE);
58                  }
59              }
60          });
61          validator.warningProperty().addListener(new ChangeListener<Boolean>() {
62              @Override
63              public void changed(ObservableValue<? extends Boolean> observableValue, Boolean aBoolean, Boolean aBoolean2) {
64                  if (aBoolean2) {
65                      addStyle(FXForm.WARNING_STYLE);
66                  } else {
67                      removeStyle(FXForm.WARNING_STYLE);
68                  }
69              }
70          });
71      }
72  
73      protected void addStyle(String styleSuffix) {
74          if (labelController.getNode() != null) {
75              labelController.getNode().getNode().getStyleClass().add(NodeType.LABEL.getStyle() + styleSuffix);
76          }
77          if (editorController.getNode() != null) {
78              editorController.getNode().getNode().getStyleClass().add(NodeType.EDITOR.getStyle() + styleSuffix);
79          }
80          if (tooltipController.getNode() != null) {
81              tooltipController.getNode().getNode().getStyleClass().add(NodeType.TOOLTIP.getStyle() + styleSuffix);
82          }
83          if (constraintController.getNode() != null) {
84              constraintController.getNode().getNode().getStyleClass().add(NodeType.CONSTRAINT.getStyle() + styleSuffix);
85          }
86      }
87  
88      protected void removeStyle(String styleSuffix) {
89          if (labelController.getNode() != null) {
90              labelController.getNode().getNode().getStyleClass().remove(NodeType.LABEL.getStyle() + styleSuffix);
91          }
92          if (editorController.getNode() != null) {
93              editorController.getNode().getNode().getStyleClass().remove(NodeType.EDITOR.getStyle() + styleSuffix);
94          }
95          if (tooltipController.getNode() != null) {
96              tooltipController.getNode().getNode().getStyleClass().remove(NodeType.TOOLTIP.getStyle() + styleSuffix);
97          }
98          if (constraintController.getNode() != null) {
99              constraintController.getNode().getNode().getStyleClass().remove(NodeType.CONSTRAINT.getStyle() + styleSuffix);
100         }
101     }
102 
103     @Override
104     protected NodeController createEditorController(AbstractFXForm fxForm, Element element) {
105         return new PropertyEditorController(fxForm, element);
106     }
107 
108     @Override
109     protected void updateSkin(FXFormSkin skin) {
110         super.updateSkin(skin);
111         if (constraintController != null) {
112             constraintController.setNode(skin.getConstraint(element));
113         }
114     }
