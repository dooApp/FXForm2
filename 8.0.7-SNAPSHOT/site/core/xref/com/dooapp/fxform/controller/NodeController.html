15  import com.dooapp.fxform.AbstractFXForm;
16  import com.dooapp.fxform.FXForm;

20  import javafx.beans.property.ObjectProperty;

23  import javafx.beans.value.ObservableValue;

26   * Created at 27/09/12 15:33.<br>

30  public abstract class NodeController implements Disposable {
31  
32      private final ObjectProperty<FXFormNode> node = new SimpleObjectProperty<FXFormNode>();
33      private final ChangeListener<FXFormNode> changeListener;
34      private final Element element;
35      private final AbstractFXForm fxForm;
36  
37      public NodeController(AbstractFXForm fxForm, Element element) {
38          this.element = element;
39          this.fxForm = fxForm;
40          changeListener = new ChangeListener<FXFormNode>() {
41              public void changed(ObservableValue<? extends FXFormNode> observableValue, FXFormNode fxFormNode, FXFormNode fxFormNode1) {
42                  if (fxFormNode != null) {
43                      unbind(fxFormNode);
44                      fxFormNode.dispose();
45                  }
46                  if (fxFormNode1 != null) {
47                      fxFormNode1.init(NodeController.this.element);
48                      bind(fxFormNode1);
49                  }
50              }
51          };
52          node.addListener(changeListener);
53      }
54  
55      protected abstract void bind(FXFormNode fxFormNode);
56  
57      protected void unbind(FXFormNode fxFormNode) {
58          fxFormNode.getProperty().unbind();
59      }
60  
61      public void dispose() {
62          setNode(null);
63          node.removeListener(changeListener);
64      }
65  
66      public FXFormNode getNode() {
67          return node.get();
68      }
69  
70      public void setNode(FXFormNode node1) {
71          node.set(node1);
72      }
73  
74      public ObjectProperty<FXFormNode> nodeProperty() {
75          return node;
76      }
77  
78      protected Element getElement() {
79          return element;
80      }
81  
82      protected AbstractFXForm getFxForm() {
83          return fxForm;
84      }
85  }
