33  public class LabelController extends NodeController {
34  
35      public final static Logger logger = Logger.getLogger(LabelController.class.getName());
36  
37      private final NodeType nodeType;
38  
39      public LabelController(AbstractFXForm fxForm, Element element, NodeType nodeType) {
40          super(fxForm, element);
41          this.nodeType = nodeType;
42      }
43  
44      @Override
45      protected void bind(FXFormNode fxFormNode) {
46          final Adapter adapter = getFxForm().getAdapterProvider().getAdapter(StringProperty.class, getNode().getProperty().getClass(), getElement(), getNode());
47          final StringProperty value = getFxForm().getResourceProvider().getString(getElement(), nodeType);
48          fxFormNode.getProperty().bind(new ObjectBinding() {
49              @Override
50              protected Object computeValue() {
51                  {
52                      bind(value);
53                  }
54  
55                  try {
56                      return adapter.adaptTo(value.get());
57                  } catch (AdapterException e) {
58                      logger.log(Level.WARNING, e.getMessage(), e);
59                  }
60                  return null;
61              }
62          });
63      }
