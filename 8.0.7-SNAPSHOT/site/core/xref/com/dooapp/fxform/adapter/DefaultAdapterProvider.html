30  public class DefaultAdapterProvider implements AdapterProvider {
31  
32      private final static Logger logger = Logger.getLogger(DefaultAdapterProvider.class.getName());
33  
34      private final static Map<AdapterMatcher, Adapter> DEFAULT_MAP = new LinkedHashMap();
35  
36      private final static Map<AdapterMatcher, Adapter> GLOBAL_MAP = new LinkedHashMap();
37  
38      private final Map<AdapterMatcher, Adapter> USER_MAP = new LinkedHashMap();
39  
40      public DefaultAdapterProvider() {
41          DEFAULT_MAP.put(new AdapterMatcher() {
42              @Override
43              public boolean matches(Class fromClass, Class toClass, Element element, FXFormNode fxFormNode) {
44                  return fromClass.isAssignableFrom(toClass);
45              }
46          }, new DefaultAdapter());
47  
48          DEFAULT_MAP.put(new TypeAdapterMatcher(IntegerProperty.class, StringProperty.class),
49                  new ConverterWrapper(new IntegerStringConverter()));
50          DEFAULT_MAP.put(new TypeAdapterMatcher(FloatProperty.class, StringProperty.class),
51                  new ConverterWrapper(new FloatStringConverter()));
52          DEFAULT_MAP.put(new TypeAdapterMatcher(LongProperty.class, StringProperty.class),
53                  new ConverterWrapper(new LongStringConverter()));
54          DEFAULT_MAP.put(new TypeAdapterMatcher(DoubleProperty.class, StringProperty.class),
55                  new ConverterWrapper(new DoubleStringConverter()));
56          DEFAULT_MAP.put(new TypeAdapterMatcher(BooleanProperty.class, StringProperty.class),
57                  new ConverterWrapper(new BooleanStringConverter()));
58          DEFAULT_MAP.put(new ObjectPropertyAdapterMatcher(BigDecimal.class, StringProperty.class),
59                  new ConverterWrapper(new BigDecimalStringConverter()));
60          DEFAULT_MAP.put(new TypeAdapterMatcher(IntegerProperty.class, DoubleProperty.class),
61                  new Adapter<Integer, Double>() {
62  
63                      @Override
64                      public Double adaptTo(Integer from) {
65                          return from.doubleValue();
66                      }
67  
68                      @Override
69                      public Integer adaptFrom(Double to) {
70                          return to.intValue();
71                      }
72                  });
73          DEFAULT_MAP.put(new TypeAdapterMatcher(FloatProperty.class, DoubleProperty.class),
74                  new Adapter<Float, Double>() {
75  
76                      @Override
77                      public Double adaptTo(Float from) {
78                          return from.doubleValue();
79                      }
80  
81                      @Override
82                      public Float adaptFrom(Double to) {
83                          return to.floatValue();
84                      }
85                  });
86  
87          DEFAULT_MAP.put(new PropertyTypeMatcher(StringProperty.class), new DefaultAdapter());
88          DEFAULT_MAP.put(new PropertyTypeMatcher(IntegerProperty.class), new DefaultAdapter());
89          DEFAULT_MAP.put(new PropertyTypeMatcher(FloatProperty.class), new DefaultAdapter());
90          DEFAULT_MAP.put(new PropertyTypeMatcher(DoubleProperty.class), new DefaultAdapter());
91          DEFAULT_MAP.put(new PropertyTypeMatcher(BooleanProperty.class), new DefaultAdapter());
92  
93          DEFAULT_MAP.put(new TypeAdapterMatcher(ObjectProperty.class, StringProperty.class),
94                  new ToStringConverter());
95      }
96  
97      @Override
98      public Adapter getAdapter(Class fromClass, Class toClass, Element element, FXFormNode fxFormNode) {
99          // check user defined factories
100         Adapter adapter = getAdapter(fromClass, toClass, element, fxFormNode, USER_MAP);
101         // check user defined global factories
102         if (adapter == null) {
103             adapter = getAdapter(fromClass, toClass, element, fxFormNode, GLOBAL_MAP);
104         }
105         // check default map
106         if (adapter == null) {
107             adapter = getAdapter(fromClass, toClass, element, fxFormNode, DEFAULT_MAP);
108         }
109         if (adapter == null) {
110             // we are converting to a String, use a generic toString converter
111             if (StringProperty.class.isAssignableFrom(toClass)) {
112                 adapter = new ToStringConverter();
113             } else {
114                 adapter = new DefaultAdapter();
115                 logger.log(java.util.logging.Level.WARNING, "No adapter between types " + fromClass + " and " + toClass + " was found (to adapt " + element + " and " + fxFormNode + ")" +
116                         "\nMake sure to register the required adapter in DefaultAdapterProvider either in the global or in the user map. See FXForm#setAdapterProvider");
117 
118             }
119         }
120         return adapter;
121     }
122 
123     private Adapter getAdapter(Class fromClass, Class toClass, Element element, FXFormNode fxFormNode, Map<AdapterMatcher, Adapter> map) {
124         for (AdapterMatcher matcher : map.keySet()) {
125             if (matcher.matches(fromClass, toClass, element, fxFormNode)) {
126                 return map.get(matcher);
127             }
128         }
129         return null;
130     }
131 
132     public static void addGlobalAdapter(AdapterMatcher adapterMatcher, Adapter adapter) {
133         GLOBAL_MAP.put(adapterMatcher, adapter);
134     }
135 
136     public void addAdapter(AdapterMatcher adapterMatcher, Adapter adapter) {
137         USER_MAP.put(adapterMatcher, adapter);
138     }
