51      @Test
52      public void testExtractExpectedFields()
53              throws
54              FormException, IllegalArgumentException {
55          Bean1 bean1 = new Bean1();
56          Bean2 bean2 = new Bean2();
57          ReflectionFieldProvider reflectionFieldProvider = new ReflectionFieldProvider();
58          List<String> include = new ArrayList<String>();
59          include.add(Bean1.class.getName() + "-property");
60          List<Field> fields = reflectionFieldProvider.getProperties(new MultipleBeanSource(bean1, bean2), include);
61          Assert.assertEquals(1, fields.size());
62          Assert.assertEquals(fields.get(0), ReflectionUtils.getFieldByName(Bean1.class, "property"));
63      }

65      @Test
66      public void testMultipleBeanSource() throws IllegalArgumentException {
67          FXForm fxForm = new FXForm();
68          Bean1 bean1 = new Bean1();
69          Bean2 bean2 = new Bean2();
70          fxForm.setSource(new MultipleBeanSource(bean1, bean2));
71          Assert.assertEquals(3, fxForm.getElements().size());
72      }

74      @Test
75      public void testMultipleBeanSourceWithIncludeFilter() throws IllegalArgumentException {
76          FXForm fxForm = new FXForm();
77          Bean1 bean1 = new Bean1();
78          Bean2 bean2 = new Bean2();
79          fxForm.getFilters().add(new IncludeFilter(bean1.getClass().getName() + "-property"));
80          fxForm.setSource(new MultipleBeanSource(bean1, bean2));
81          Assert.assertEquals(1, fxForm.getFilteredElements().size());
82          Assert.assertEquals(fxForm.getFilteredElements().get(0).getDeclaringClass(), Bean1.class);
83          fxForm.getFilters().clear();
84          fxForm.getFilters().add(new IncludeFilter(bean2.getClass().getName() + "-property"));
85          Assert.assertEquals(1, fxForm.getFilteredElements().size());
86          Assert.assertEquals(fxForm.getFilteredElements().get(0).getDeclaringClass(), Bean2.class);
87      }

89      @Test
90      public void testMultipleBeanSourceWithExcludeFilter() throws IllegalArgumentException {
91          FXForm fxForm = new FXForm();
92          Bean1 bean1 = new Bean1();
93          Bean2 bean2 = new Bean2();
94          fxForm.getFilters().add(new ExcludeFilter(bean1.getClass().getName() + "-property"));
