40  public class DefaultFactoryProvider implements FactoryProvider {
41  
42      private final static Map<ElementHandler, Callback<Void, FXFormNode>> DEFAULT_MAP = new LinkedHashMap<ElementHandler, Callback<Void, FXFormNode>>();
43  
44      private final static Map<ElementHandler, Callback<Void, FXFormNode>> GLOBAL_MAP = new LinkedHashMap<ElementHandler, Callback<Void, FXFormNode>>();
45  
46      private final Map<ElementHandler, Callback<Void, FXFormNode>> USER_MAP = new LinkedHashMap<ElementHandler, Callback<Void, FXFormNode>>();
47  
48      public DefaultFactoryProvider() {
49          // register default delegates
50          DEFAULT_MAP.put(new TypeFieldHandler(StringProperty.class), new TextFieldFactory());
51          DEFAULT_MAP.put(new TypeFieldHandler(BooleanProperty.class), new CheckboxFactory());
52          DEFAULT_MAP.put(new EnumHandler(), new EnumChoiceBoxFactory());
53          DEFAULT_MAP.put(new TypeFieldHandler(IntegerProperty.class), new TextFieldFactory());
54          DEFAULT_MAP.put(new TypeFieldHandler(LongProperty.class), new TextFieldFactory());
55          DEFAULT_MAP.put(new TypeFieldHandler(DoubleProperty.class), new TextFieldFactory());
56          DEFAULT_MAP.put(new TypeFieldHandler(ListProperty.class), new TableViewFactory());
57          DEFAULT_MAP.put(new WrappedTypeHandler(BigDecimal.class), new TextFieldFactory());
58          DEFAULT_MAP.put(new WrappedTypeHandler(Color.class), new ColorPickerFactory());
59          DEFAULT_MAP.put(new WrappedTypeHandler(LocalDate.class), new DatePickerFactory());
60          DEFAULT_MAP.put(new TypeFieldHandler(ReadOnlyStringProperty.class), new LabelFactory());
61          DEFAULT_MAP.put(new TypeFieldHandler(ReadOnlyBooleanProperty.class), new CheckboxFactory());
62          DEFAULT_MAP.put(new TypeFieldHandler(ReadOnlyIntegerProperty.class), new LabelFactory());
63          DEFAULT_MAP.put(new TypeFieldHandler(ReadOnlyLongProperty.class), new LabelFactory());
64          DEFAULT_MAP.put(new TypeFieldHandler(ReadOnlyDoubleProperty.class), new LabelFactory());
65      }
66  
67      private Callback<Void, FXFormNode> getDelegate(Element element, Map<ElementHandler, Callback<Void, FXFormNode>> map) {
68          for (ElementHandler handler : map.keySet()) {
69              if (handler.handle(element)) {
70                  return map.get(handler);
71              }
72          }
73          return null;
74      }
75  
76      public static void addGlobalFactory(ElementHandler handler, Callback<Void, FXFormNode> factory) {
77          GLOBAL_MAP.put(handler, factory);
78      }
79  
80      public void addFactory(ElementHandler handler, Callback<Void, FXFormNode> factory) {
81          USER_MAP.put(handler, factory);
82      }
83  
84      /**
85       * Create the node by trying to find a delegate factory.
86       * This method will lookup in the user map, the global map and finally in the default map.
87       *
88       * @return the created node
89       * @throws NodeCreationException
90       */
91      public Callback<Void, FXFormNode> getFactory(Element element) {
92          // check user defined factories
93          Callback<Void, FXFormNode> delegate = getDelegate(element, USER_MAP);
94          // check user defined global factories
95          if (delegate == null) {
96              delegate = getDelegate(element, GLOBAL_MAP);
97          }
98          // check default map
99          if (delegate == null) {
100             delegate = getDelegate(element, DEFAULT_MAP);
101         }
102         return delegate;
103     }

105     public String toString() {
106         return "[DefaultFactoryProvider\n"
107                 + "GLOBAL_MAP:\n" + dumpMap(GLOBAL_MAP)
108                 + "\nUSER_MAP:\n" + dumpMap(USER_MAP)
109                 + "]";
110     }
111 
112     private String dumpMap(Map map) {
113         StringBuilder sb = new StringBuilder();
114         Iterator<Map.Entry> iter = map.entrySet().iterator();
115         while (iter.hasNext()) {
116             Map.Entry entry = iter.next();
117             sb.append(entry.getKey());
118             sb.append('=').append('"');
119             sb.append(entry.getValue());
120             sb.append('"');
121             if (iter.hasNext()) {
122                 sb.append(',').append('\n');
123             }
124         }
125         return sb.toString();
126     }
