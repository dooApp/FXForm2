33  				String label;
34  				try {
35  					label = getResourceBundle().getString(element.getDeclaringClass().getName()
36  							+ "-" + element.getName()
37  							+ "-" + nodeType.name().toLowerCase());
38  				} catch (Exception e1) {
39  					try {
40  						label = getResourceBundle().getString(element.getDeclaringClass().getSimpleName()
41  								+ "-" + element.getName()
42  								+ "-" + nodeType.name().toLowerCase());
43  					} catch (Exception e2) {
44  						// Look for a generic label without the bean name
45  						try {
46  							label = getResourceBundle().getString(element.getName()
47  									+ "-" + nodeType.name().toLowerCase());
48  						} catch (Exception e3) {
49  							// Label is undefined in the resource bundle
50  							label = handleDefaultValue(element.getName(), nodeType);
51  						}
52  					}
53  				}
54  				return label;
55  			}
56  		});
57  		return string;
58  	}
